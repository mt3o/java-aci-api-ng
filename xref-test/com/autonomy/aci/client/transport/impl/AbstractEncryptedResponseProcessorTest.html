31      public void testFailToCreateXMLStreamReader() throws NoSuchFieldException, XMLStreamException, IllegalAccessException {
32          final AbstractEncryptedResponseProcessorImpl processor = new AbstractEncryptedResponseProcessorImpl();
33  
34          final Field field = ReflectionTestUtils.getAccessibleField(AbstractEncryptedResponseProcessor.class, "xmlInputFactory");
35          field.set(processor, when(mock(XMLInputFactory.class).createXMLStreamReader(any(InputStream.class))).thenThrow(XMLStreamException.class).getMock());

37          processor.process(mock(InputStream.class));
38          fail("Should have thrown a ProcessorException...");
39      }

41      @Test
42      @SuppressWarnings("unchecked")
43      public void testXMLStreamReaderClosed() throws NoSuchFieldException, XMLStreamException, IllegalAccessException {
44          final AbstractEncryptedResponseProcessorImpl processor = new AbstractEncryptedResponseProcessorImpl();
45  
46          final Field field = ReflectionTestUtils.getAccessibleField(AbstractEncryptedResponseProcessor.class, "xmlInputFactory");
47          final XMLStreamReader mockXmlStreamReader = when(mock(XMLStreamReader.class).hasNext()).thenThrow(ProcessorException.class).getMock();
48          field.set(processor, when(mock(XMLInputFactory.class).createXMLStreamReader(any(InputStream.class))).thenReturn(mockXmlStreamReader).getMock());
49  
50          try {
51              processor.process(mock(InputStream.class));

54          }
55      }

59      public void testXMLStreamReaderErrorOnClose() throws NoSuchFieldException, XMLStreamException, IllegalAccessException {
60          final AbstractEncryptedResponseProcessorImpl processor = new AbstractEncryptedResponseProcessorImpl();
61  
62          final Field field = ReflectionTestUtils.getAccessibleField(AbstractEncryptedResponseProcessor.class, "xmlInputFactory");
63          final XMLStreamReader mockXmlStreamReader = when(mock(XMLStreamReader.class).hasNext()).thenThrow(ProcessorException.class).getMock();
