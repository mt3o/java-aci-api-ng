25  /**
26   * Abstract <tt>Processor</tt> that should be used by all processors wanting to process the ACI response via the
27   * <a href="http://jcp.org/en/jsr/detail?id=173">JSR173</a> StAX API. Some of the properties that can be set on the
28   * {@link XMLInputFactory} are exposed, so that the implementation can be modified to suit the required situation.
29   * For example, setting {@link #namespaceAware} and {@link #validating} to <tt>false</tt> should theoretically speed
30   * up parsing, especially as ACI responses shouldn't require either to be parsed, (both are <tt>false</tt> by default
31   * for this very reason). When a subclass is created, this class checks to see if any of the properties have been set
32   * as system properties and if they have it sets the corresponding property to this value, these values can then be
33   * overridden by using the appropriate setter method.
34   */

81      /**
82       * This constructor gets a new {@link XMLInputFactory} instance that is reused every time
83       * {@link #process(com.autonomy.aci.client.transport.AciResponseInputStream)} is called, this
84       * should be faster than creating a new instance every time this method is called.
85       * <p/>
86       * The properties are set to the following defaults if they are not specified as system properties:
87       * <table>
88       * <tr><th>Property</th><th>Default</th></tr>
89       * <tr><td>XMLInputFactory.IS_NAMESPACE_AWARE</td><td><tt>false</tt></td></tr>
90       * <tr><td>XMLInputFactory.IS_VALIDATING</th><th><tt>false</tt></td></tr>
91       * <tr><td>XMLInputFactory.IS_COALESCING</th><th><tt>false</tt></td></tr>
92       * <tr><td>XMLInputFactory.IS_REPLACING_ENTITY_REFERENCES</th><th><tt>true</tt></td></tr>
93       * <tr><td>XMLInputFactory.IS_SUPPORTING_EXTERNAL_ENTITIES</th><th><tt>false</tt></td></tr>
94       * <tr><td>XMLInputFactory.SUPPORT_DTD</th><th><tt>true</tt></td></tr>
95       * </table>
96       */
97      protected AbstractStAXProcessor() {
98          // See if the various XMLInputFactory properties are set as system properties...
99          namespaceAware = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.IS_NAMESPACE_AWARE), "false"));
100         validating = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.IS_VALIDATING), "false"));
101         coalescing = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.IS_COALESCING), "false"));
102         replacingEntityReferences = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.IS_REPLACING_ENTITY_REFERENCES), "true"));
103         supportingExternalEntities = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.IS_SUPPORTING_EXTERNAL_ENTITIES), "false"));
104         supportDtd = BooleanUtils.toBoolean(StringUtils.defaultString(System.getProperty(XMLInputFactory.SUPPORT_DTD), "true"));

106         // Create the XMLStreamReader factory...

118     /**
119      * This method firstly checks that the content type of the response is text based and can be parsed. If so, it
120      * converts the <tt>AciResponseInputStream</tt> into a StAX <tt>XMLStreamReader</tt> and calls the the {@link
121      * #process(javax.xml.stream.XMLStreamReader)} method that should be implemented in a subclass to do all the work.
122      * @param aciResponseInputStream The ACI response to process
123      * @return An object of type <tt>T</tt>
124      * @throws AciErrorException  If the ACI response was an error response
125      * @throws ProcessorException If an error occurred during the processing of the IDOL response
126      */
127     public T process(final AciResponseInputStream aciResponseInputStream) {
128         LOGGER.trace("process() called...");
129 
130         if (!"text/xml".equalsIgnoreCase(aciResponseInputStream.getContentType())) {
131             throw new ProcessorException("This processor is unable to process non-text ACI responses. The content type for this response is " + aciResponseInputStream.getContentType());
132         }
133 
134         // Define this here so we can make sure it's closed when the processor is finished...
135         XMLStreamReader xmlStreamReader = null;
136 
137         try {
138             // Update the factory with the various properties as they might have changed since the last run...
139             xmlInputFactory.setProperty(XMLInputFactory.IS_NAMESPACE_AWARE, namespaceAware);
140             xmlInputFactory.setProperty(XMLInputFactory.IS_VALIDATING, validating);
141             xmlInputFactory.setProperty(XMLInputFactory.IS_COALESCING, coalescing);
142             xmlInputFactory.setProperty(XMLInputFactory.IS_REPLACING_ENTITY_REFERENCES, replacingEntityReferences);
143             xmlInputFactory.setProperty(XMLInputFactory.IS_SUPPORTING_EXTERNAL_ENTITIES, supportingExternalEntities);
144             xmlInputFactory.setProperty(XMLInputFactory.SUPPORT_DTD, supportDtd);
145 
146             // Convert the input stream..
147             xmlStreamReader = xmlInputFactory.createXMLStreamReader(aciResponseInputStream);
148             return process(xmlStreamReader);
149         } catch (final XMLStreamException xmlse) {
150             throw new ProcessorException("Unable to convert the InputStream to a XMLStreamReader", xmlse);
151         } finally {
152             if (xmlStreamReader != null) {
153                 try {
154                     // This does NOT close the underlying AciResponseInputStream
155                     xmlStreamReader.close();
156                 } catch (final XMLStreamException xmlse) {
157                     LOGGER.error("Unable to close the XMLStreamReader.", xmlse);
158                 }
159             }
160         }
161     }

172     /**
173      * Reads from the XML stream and tries to determine if the ACI response contains an error or not. The stream is left
174      * at the end of the body content of the <tt>/autnresponse/response</tt> element, if one could be found.
175      * @param xmlStreamReader The response to process
176      * @return <tt>true</tt> if the response contains an error, <tt>false</tt> otherwise
177      * @throws javax.xml.stream.XMLStreamException If there was a problem reading the IDOL Server response.
178      */
179     protected boolean isErrorResponse(final XMLStreamReader xmlStreamReader) throws XMLStreamException {
180         LOGGER.trace("isErrorResponse() called...");
181 
182         // Get the /autnresponse/response element...
183         while (xmlStreamReader.hasNext()) {
184             // Get the event type...
185             final int eventType = xmlStreamReader.next();
186 
187             // Check to see if it's a start event...
188             if ((XMLEvent.START_ELEMENT == eventType) && ("response".equalsIgnoreCase(xmlStreamReader.getLocalName()))) {
189                 return "ERROR".equalsIgnoreCase(xmlStreamReader.getElementText());
190             }
191         }
192 
193         // Couldn't find a /autnresponse/response element...
194         throw new XMLStreamException("Unable to find /autnresponse/response element.");
195     }

197     /**
198      * Process the remainder of the IDOL response with the configured error processor.
199      * @param xmlStreamReader The IDOL response fragment containing the error information
200      * @throws AciErrorException        Once the response has been parsed for all the error information it contains
201      * @throws ProcessorException       If something went wrong while trying to process the error response
202      * @throws IllegalArgumentException if no error processor has been set.
203      */
204     protected void processErrorResponse(final XMLStreamReader xmlStreamReader) {
205         LOGGER.trace("processErrorResponse() ");
206 
207         // Sanity check...
208         Validate.notNull(errorProcessor, "Unable to process the error response, as no errorProcessor has been configured.");
209 
210         // Process the error response and propagate the resulting exception...
211         errorProcessor.process(xmlStreamReader);
212     }

214     /**
215      * Move the cursor forward through the XML stream to the next start element.
216      * @param xmlStreamReader The XML stream to use
217      * @throws XMLStreamException If there was an error using the stream
218      */
219     protected void forwardToNextStartElement(final XMLStreamReader xmlStreamReader) throws XMLStreamException {
220         while (xmlStreamReader.hasNext()) {
221             final int eventType = xmlStreamReader.next();
222             if (XMLEvent.START_ELEMENT == eventType) {
223                 return;
224             }
225         }
226         throw new XMLStreamException("No more START_ELEMENT events found");
227     }

229     /**
230      * Move the cursor forward through the XML stream to the next start or end element, which ever comes first.
231      * @param xmlStreamReader The XML stream to use
232      * @return The type of event forwarded to, i.e. <tt>XMLEvent.START_ELEMENT</tt> or <tt>XMLEvent.END_ELEMENT</tt>.
233      * @throws XMLStreamException If there was an error using the stream
234      */
235     protected int forwardToNextStartOrEndElement(final XMLStreamReader xmlStreamReader) throws XMLStreamException {
236         while (xmlStreamReader.hasNext()) {
237             final int eventType = xmlStreamReader.next();
238             if ((XMLEvent.START_ELEMENT == eventType) || (XMLEvent.END_ELEMENT == eventType)) {
239                 return eventType;
240             }
241         }
242         throw new XMLStreamException("No more START_ELEMENT or END_ELEMENT events found");
243     }

245     /**
246      * Forwards through the stream looking for the an element with <tt>elementName</tt>
247      * @param elementName     The name of the element to find.
248      * @param xmlStreamReader The stream to forward through
249      * @throws XMLStreamException If there was an error using the stream, or if no element with <tt>elementName</tt>
250      *                            could be found
251      */
252     protected void forwardToNamedStartElement(final String elementName, final XMLStreamReader xmlStreamReader) throws XMLStreamException {
253         while (xmlStreamReader.hasNext()) {
254             final int eventType = xmlStreamReader.next();
255             if ((eventType == XMLEvent.START_ELEMENT) && (elementName.equals(xmlStreamReader.getLocalName()))) {
256                 return;
257             }
258         }
259         throw new XMLStreamException("Unable to find a start element for, " + elementName);
260     }
