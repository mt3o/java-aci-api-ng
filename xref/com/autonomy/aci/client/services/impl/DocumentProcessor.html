29  /**
30   * <tt>Processor</tt> implementation that converts an ACI response into a DOM <tt>Document</tt> for further processing
31   * either directly via DOM or via XPath.
32   */
33  public class DocumentProcessor implements Processor<Document> {
34  
35      private static final long serialVersionUID = -1757174558649421058L;
36  
37      private static final Logger LOGGER = LoggerFactory.getLogger(DocumentProcessor.class);
38  
39      /**
40       * Converts the <tt>InputStream</tt> into a DOM <tt>Document</tt>.
41       * @param response The <tt>InputStream to convert</tt>
42       * @return The resulting DOM <tt>Document</tt>
43       * @throws ProcessorException If there was an exception while parsing the input stream or configuring the
44       *                            <tt>DocumentBuilderFactory</tt>.
45       */
46      private Document convertACIResponseToDOM(final InputStream response) {
47          LOGGER.trace("convertACIResponseToDOM() called...");
48  
49          try {
50              // Create the factory...
51              final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
52  
53              // Should speed things up a bit...
54              factory.setNamespaceAware(false);
55              factory.setValidating(false);
56  
57              LOGGER.debug("Converting ACI response to DOM Document...");
58  
59              // Get a document builder and convert the document...
60              return factory.newDocumentBuilder().parse(new InputSource(response));
61          } catch (final ParserConfigurationException pce) {
62              throw new ProcessorException("Unable to parse the ACI response into a DOM document.", pce);
63          } catch (final SAXException saxe) {
64              throw new ProcessorException("Unable to parse the ACI response into a DOM document.", saxe);
65          } catch (final IOException ioe) {
66              throw new ProcessorException("Unable to parse the ACI response into a DOM document.", ioe);
67          }
68      }
69  
70      /**
71       * Checks the DOM <tt>Document</tt> to see if it is an ACI Server error response. If it is, it pulls all the
72       * information contained in the response into an <tt>AciErrorException</tt> and throws it, otherwise it does
73       * nothing.
74       * @param response The DOM <tt>Document</tt> to check.
75       * @throws AciErrorException  If an error response was detected.
76       * @throws ProcessorException If there was any other kind of exception caught during processing.
77       */
78      private void checkACIResponseForError(final Document response) {
79          LOGGER.trace("checkACIResponseForError() called...");
80  
81          // Create an xpath object for getting stuff with...
82          final XPath xpath = XPathFactory.newInstance().newXPath();
83  
84          try {
85              // Get the value of the <response> tag...
86              final String value = xpath.evaluate("/autnresponse/response", response);
87  
88              LOGGER.debug("Response tag has value - {}...", value);
89  
90              if ("ERROR".equals(value)) {
91                  LOGGER.debug("Error response detected, creating an AciErrorException...");
92  
93                  // Create an exception to throw...
94                  final AciErrorException error = new AciErrorException();
95  
96                  // Get the error properties...
97                  error.setErrorId(xpath.evaluate("/autnresponse/responsedata/error/errorid", response));
98                  error.setRawErrorId(xpath.evaluate("/autnresponse/responsedata/error/rawerrorid", response));
99                  error.setErrorString(xpath.evaluate("/autnresponse/responsedata/error/errorstring", response));
100                 error.setErrorDescription(xpath.evaluate("/autnresponse/responsedata/error/errordescription", response));
101                 error.setErrorCode(xpath.evaluate("/autnresponse/responsedata/error/errorcode", response));
102                 try {
103                     error.setErrorTime(DateTimeUtils.getInstance().parseDate(xpath.evaluate("/autnresponse/responsedata/error/errortime", response), "dd MMM yy HH:mm:ss"));
104                 } catch (final ParseException pe) {
105                     LOGGER.error("ParseException caught while trying to convert errortime tag into java.util.Date.", pe);
106                 }
107 
108                 // Throw the error...
109                 throw error;
110             }
111         } catch (final XPathExpressionException xpee) {
112             // Throw a wobbler, as this should never happen...
113             throw new ProcessorException("XPathExpressionException caught while trying to check ACI response for ERROR flag.", xpee);
114         }
115     }

117     /**
118      * Process the ACI response input into a DOM <tt>Document</tt>.
119      * @param aciResponse The ACI response to process
120      * @return A DOM <tt>Document</tt>
121      * @throws AciErrorException  If the ACI response was an error response
122      * @throws ProcessorException If an error occurred during the processing of the ACI response
123      */
124     public Document process(final AciResponseInputStream aciResponse) {
125         LOGGER.trace("process() called...");
126 
127         // Convert the stream into a DOM Document...
128         final Document document = convertACIResponseToDOM(aciResponse);
129 
130         LOGGER.debug("Checking for ERROR response...");
131 
132         // Check the response for an error flag...
133         checkACIResponseForError(document);
134 
135         LOGGER.debug("Returning the DOM Document to caller...");
136 
137         // Return the document...
138         return document;
139     }
