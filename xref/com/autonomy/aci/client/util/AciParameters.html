19  /**
20   * Utility class to make it easier to specify sets of <tt>AciParameter</tt> objects for sending with ACI actions. In the
21   * past you had to do something like:
22   * <pre>
23   *    Set&lt;AciParameter&gt; parameters = new LinkedHashSet&lt;AciParameter&gt;();
24   *    parameters.add(new AciParameter(&quot;action&quot;, &quot;query&quot;));
25   *    parameters.add(new AciParameter(&quot;text&quot;, &quot;XPath&quot;));
26   *    parameters.add(new AciParameter(&quot;combine&quot;, &quot;simple&quot;));
27   *    ...
28   * </pre>
29   * now you can do:
30   * <pre>
31   *    AciParameters parameters = new AciParameters();
32   *    parameters.add(&quot;action&quot;, &quot;query&quot;);
33   *    parameters.add(&quot;text&quot;, &quot;XPath&quot;);
34   *    parameters.add(&quot;combine&quot;, &quot;simple&quot;);
35   *    ...
36   * </pre>
37   * or:
38   * <pre>
39   *    AciParameters parameters = new AciParameters(
40   *            new AciParameter(&quot;action&quot;, &quot;query&quot;),
41   *            new AciParameter(&quot;text&quot;, &quot;XPath&quot;),
42   *            new AciParameter(&quot;combine&quot;, &quot;simple&quot;)
43   *    );
44   *    ...
45   * </pre>
46   * For those actions that don't generally take parameters, like <tt>GetStatus</tt> and <tt>GetVersion</tt>, you can do
47   * the following:
48   * <pre>
49   *    ...
50   *    aciService.executeAction(new AciParameters(&quot;GetStatus&quot;), myProcessor);
51   *    ...
52   * </pre>
53   */

212     /**
213      * Puts the specified <tt>parameter</tt> into this collection. If a parameter with the same name already exists in
214      * this collection, it is replaced by the specified <tt>parameter</tt>.
215      * @param parameter The parameter to put into this collection
216      * @return Previous parameter or <tt>null</tt> if the parameter didn't exist
217      * @throws IllegalArgumentException If <tt>parameter</tt> is <tt>null</tt>
218      */
219     public AciParameter put(final AciParameter parameter) {
220         Validate.notNull(parameter, "The parameter must not be null.");
221 
222         // Find the existing parameter...
223         AciParameter oldParameter = null;
224         for (final AciParameter param : parameters) {
225             if (param.equals(parameter)) {
226                 oldParameter = param;
227                 break;
228             }
229         }
230 
231         // Remove the old parameter if there was one...
232         if (oldParameter != null) {
233             parameters.remove(oldParameter);
234         }
235 
236         // Put in the new parameter...
237         parameters.add(parameter);
238 
239         // Return what ever the old parameter was...
240         return oldParameter;
241     }
