20  /**
21   * This is a utility class to help create and configure an instance of Apache {@link HttpClient} as it appears almost
22   * impossible to do so via an IoC container like Spring. This factory assumes that you'll be running in a multi-threaded

86      /**
87       * Creates an instance of <tt>DefaultHttpClient</tt> with a <tt>ThreadSafeClientConnManager</tt>.
88       * @return an implementation of the <tt>HttpClient</tt> interface.
89       */
90      public HttpClient createInstance() {
91          LOGGER.debug("Creating a new instance of DefaultHttpClient with configuration -> {}", toString());
92  
93          // Create the connection manager which will be default create the necessary schema registry stuff...
94          final PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager();
95          connectionManager.setMaxTotal(maxTotalConnections);
96          connectionManager.setDefaultMaxPerRoute(maxConnectionsPerRoute);
97  
98          // Set the HTTP connection parameters (These are in the HttpCore JavaDocs, NOT the HttpClient ones)...
99          final HttpParams params = new BasicHttpParams();
100         HttpConnectionParams.setConnectionTimeout(params, connectionTimeout);
101         HttpConnectionParams.setLinger(params, linger);
102         HttpConnectionParams.setSocketBufferSize(params, socketBufferSize);
103         HttpConnectionParams.setSoKeepalive(params, soKeepAlive);
104         HttpConnectionParams.setSoReuseaddr(params, soReuseAddr);
105         HttpConnectionParams.setSoTimeout(params, soTimeout);
106         HttpConnectionParams.setStaleCheckingEnabled(params, staleCheckingEnabled);
107         HttpConnectionParams.setTcpNoDelay(params, tcpNoDelay);
108 
109         // Create the HttpClient and configure the compression interceptors if required...
110         final DefaultHttpClient httpClient = new DefaultHttpClient(connectionManager, params);
111 
112         if (useCompression) {
113             httpClient.addRequestInterceptor(new RequestAcceptEncoding());
114             httpClient.addResponseInterceptor(new DeflateContentEncoding());
115         }
116 
117         return httpClient;
118     }
