44  /**
45   * Implementation of the {@link com.autonomy.aci.client.transport.AciHttpClient} interface that provides the actual HTTP
46   * communication mechanism. This implementation uses the HttpClient provided by the <a href="http://hc.apache.org/">
47   * Apache HttpComponents</a> project. It defaults to using the HTTP <tt>GET</tt> method, if you wish to send ACI actions
48   * with the HTTP <tt>POST</tt> method, then call the {@link #setUsePostMethod(boolean)} method with {@code true}.
49   * <p/>
50   * This implementation of the {@link com.autonomy.aci.client.transport.AciHttpClient} interface does no configuration of
51   * the {@code HttpClient} that it uses. It expects all the configuration to have been done by the user before passing it
52   * to this object. This configuration can be done in normal code, via the
53   * {@link com.autonomy.aci.client.transport.impl.HttpClientFactory}, or via an IoC container like
54   * <a href="http://www.springsource.org/">Spring</a>.
55   * @see <a href="http://hc.apache.org/">Apache HttpComponents</a>
56   */

93      /**
94       * Turns the {@code parameters} and {@code serverDetails} into either an HTTP GET or POST request.
95       * @param serverDetails The details of the ACI server the request will be sent to
96       * @param parameters    The parameters to send with the ACI action.
97       * @return A HTTP GET or POST request that can be used to execute the ACI action
98       * @throws EncryptionCodecException     If something went wrong encrypting the parameters
99       * @throws URISyntaxException           If something went wrong creating the URI to send the action to
100      * @throws UnsupportedEncodingException If there was a problem working with the parameters in the specified
101      *                                      character encoding
102      */
103     private HttpUriRequest constructHttpRequest(final AciServerDetails serverDetails, final Set<? extends AciParameter> parameters) throws EncryptionCodecException, URISyntaxException, UnsupportedEncodingException {
104         LOGGER.trace("constructHttpMethod() called...");
105 
106         // Copy the parameters...
107         final Set<? extends AciParameter> params = (serverDetails.getEncryptionCodec() == null)
108                 ? parameters
109                 : createEncryptedParameters(serverDetails, parameters);
110 
111         // Only create a POST method if required...
112         return usePostMethod ? createPostMethod(serverDetails, params) : createGetMethod(serverDetails, params);
113     }

115     /**
116      * Takes the passed in set of parameters and encrypts them.
117      * @param serverDetails The details of the ACI server the request will be sent to
118      * @param parameters    The parameters to send with the ACI action.
119      * @return A set of encrypted parameters
120      * @throws EncryptionCodecException if something went wrong encrypting the parameters
121      */
122     private Set<? extends AciParameter> createEncryptedParameters(final AciServerDetails serverDetails, final Set<? extends AciParameter> parameters) throws EncryptionCodecException {
123         LOGGER.trace("createEncryptedParameters() called...");
124 
125         // Generate the query String and put it through the codec...
126         final String data = EncryptionCodecUtils.getInstance().encrypt(
127                 serverDetails.getEncryptionCodec(),
128                 convertParameters(parameters, serverDetails.getCharsetName()),
129                 serverDetails.getCharsetName()
130         );
131 
132         // Create the parameters for an encrypted action...
133         return new AciParameters(
134                 new AciParameter(AciConstants.PARAM_ACTION, AciConstants.ACTION_ENCRYPTED),
135                 new AciParameter(AciConstants.PARAM_DATA, data)
136         );
137     }

139     /**
140      * Create a {@code GetMethod} and adds the ACI parameters to the query string.
141      * @param serverDetails The details of the ACI server the request will be sent to
142      * @param parameters    The parameters to send with the ACI action.
143      * @return a {@code HttpGet} that is ready to execute the ACI action.
144      * @throws URISyntaxException If there was a problem construction the request URI from the <tt>serverDetails</tt>
145      *                            and <tt>parameters</tt>
146      */
147     private HttpUriRequest createGetMethod(final AciServerDetails serverDetails, final Set<? extends AciParameter> parameters) throws URISyntaxException {
148         LOGGER.trace("createGetMethod() called...");
149 
150         // Create the URI to use...
151         final URI uri = new URIBuilder()
152                 .setScheme(serverDetails.getProtocol().toString().toLowerCase(Locale.ENGLISH))
153                 .setHost(serverDetails.getHost())
154                 .setPort(serverDetails.getPort())
155                 .setPath("/")
156                 .setQuery(convertParameters(parameters, serverDetails.getCharsetName()))
157                 .build();
158 
159         // Return the constructed get method...
160         return new HttpGet(uri);
161     }

163     /**
164      * Create a {@code PostMethod} and adds the ACI parameters to the request body.
165      * @param serverDetails The details of the ACI server the request will be sent to
166      * @param parameters    The parameters to send with the ACI action.
167      * @return An {@code HttpPost} that is ready to execute the ACI action.
168      * @throws UnsupportedEncodingException Will be thrown if <tt>serverDetails.getCharsetName()</tt> returns a
169      *                                      charset that is not supported by the JVM
170      * @throws URISyntaxException           If there was a problem construction the request URI from the
171      *                                      <tt>serverDetails</tt> and <tt>parameters</tt>
172      */
173     private HttpUriRequest createPostMethod(final AciServerDetails serverDetails, final Set<? extends AciParameter> parameters) throws URISyntaxException, UnsupportedEncodingException {
174         LOGGER.trace("createPostMethod() called...");
175 
176         // Create the URI to use...
177         final URI uri = new URIBuilder()
178                 .setScheme(serverDetails.getProtocol().toString().toLowerCase(Locale.ENGLISH))
179                 .setHost(serverDetails.getHost())
180                 .setPort(serverDetails.getPort())
181                 .setPath("/")
182                 .build();
183 
184         // Create the method...
185         final HttpPost method = new HttpPost(uri);
186 
187         // Convert the parameters into an entity...
188         method.setEntity(
189                 new StringEntity(
190                         convertParameters(parameters, serverDetails.getCharsetName()),
191                         serverDetails.getCharsetName()
192                 )
193         );
194 
195         // Return the method...
196         return method;
197     }

199     /**
200      * Converts a list of {@code AciParameter} objects into an array of {@code NameValuePair} objects suitable for use
201      * in both POST and GET methods.
202      * @param parameters  The set of parameters to convert.
203      * @param charsetName The name of the charset to use when encoding the parameters
204      * @return an <tt>String</tt> representing the query string portion of a URI
205      */
206     private String convertParameters(final Set<? extends AciParameter> parameters, final String charsetName) {
207         LOGGER.trace("convertParameters() called...");
208 
209         // Just incase, remove the allowed null entry...
210         parameters.remove(null);
211 
212         final List<NameValuePair> pairs = new ArrayList<NameValuePair>(parameters.size());
213 
214         LOGGER.debug("Converting {} parameters...", parameters.size());
215 
216         NameValuePair actionPair = null;
217 
218         for (final AciParameter parameter : parameters) {
219             if (AciConstants.PARAM_ACTION.equalsIgnoreCase(parameter.getName())) {
220                 actionPair = new BasicNameValuePair(parameter.getName(), parameter.getValue());
221             } else {
222                 pairs.add(new BasicNameValuePair(parameter.getName(), parameter.getValue()));
223             }
224         }
225 
226         // Ensure that the action=XXX parameter is the first thing in the list...
227         Validate.isTrue(actionPair != null, "No action parameter found in parameter set, please set one before trying to execute an ACI request.");
228         pairs.add(0, actionPair);
229 
230         // Convert to a string and return...
231         return URLEncodedUtils.format(pairs, charsetName);
232     }

234     /**
235      * Execute an ACI action on the specific ACI server.
236      * @param serverDetails Details of the ACI server to send the action to
237      * @param parameters    The parameters to send with the ACI action
238      * @return An <tt>AciResponseInputStream</tt> containing the ACI response
239      * @throws IOException              If an I/O (transport) error occurs. Some transport exceptions can be recovered from
240      * @throws AciHttpException         If a protocol exception occurs. Usually protocol exceptions cannot be recovered from
241      * @throws IllegalArgumentException if the <tt>httpClient</tt> property is <tt>null</tt> or <tt>parameters</tt> is <tt>null</tt>
242      */
243     public AciResponseInputStream executeAction(final AciServerDetails serverDetails, final Set<? extends AciParameter> parameters) throws IOException, AciHttpException {
244         LOGGER.trace("executeAction() called...");
245 
246         Validate.notNull(httpClient, "You must set the HttpClient instance to use before using this class.");
247         Validate.notEmpty(parameters, "The parameter set must not be null or empty.");
248 
249         try {
250             // Create the method we're going to use...
251             final HttpUriRequest request = constructHttpRequest(serverDetails, parameters);
252 
253             LOGGER.debug("Executing action on {}:{}...", serverDetails.getHost(), serverDetails.getPort());
254 
255             // Execute the method...
256             final HttpResponse response = httpClient.execute(request);
257 
258             final int statusCode = response.getStatusLine().getStatusCode();
259             LOGGER.debug("Executed method and got status code - {}...", statusCode);
260 
261             // Treat anything other than a 2xx status code as an error...
262             if ((statusCode < 200) || (statusCode >= 300)) {
263                 // close the connection so it can be reused
264                 EntityUtils.consume(response.getEntity());
265 
266                 throw new AciHttpException("The server returned a status code, " + statusCode + ", that wasn't in the 2xx Success range.");
267             }
268 
269             // Decorate the InputStream so we can release the HTTP connection once the stream's been read...
270             return decryptResponse(serverDetails.getEncryptionCodec(), response)
271                     ? new DecryptingAciResponseInputStreamImpl(serverDetails, response)
272                     : new AciResponseInputStreamImpl(response);
273         } catch (final ClientProtocolException cpe) {
274             throw new AciHttpException("A HTTP protocol Exception has been caught while trying to execute the ACI request.", cpe);
275         } catch (final EncryptionCodecException ece) {
276             throw new AciHttpException("Unable to send the ACI request due to an encryption failure.", ece);
277         } catch (final URISyntaxException urise) {
278             throw new AciHttpException("Unable to construct the URI required to send the ACI request.", urise);
279         }
280     }

282     private boolean decryptResponse(final EncryptionCodec encryptionCodec, final HttpResponse response) {
283         LOGGER.trace("decryptResponse() called...");
284 
285         // If there is no encryptionCodec then we don't need to check the headers...
286         boolean decryptResponse = (encryptionCodec != null);
287 
288         LOGGER.debug("Using an EncryptionCodec - {}...", decryptResponse);
289 
290         if (decryptResponse) {
291             LOGGER.debug("Checking AUTN-Content-Type response header...");
292 
293             // This response header is only supplied with encrypted responses, so if it's not there we don't decrypt,
294             // i.e. it's either an OEM IDOL, or they did encryptResponse=false...
295             final Header header = response.getFirstHeader("AUTN-Content-Type");
296             if (header == null) {
297                 LOGGER.debug("No AUTN-Content-Type response header, so don't auto decrypt response...");
298                 decryptResponse = false;
299             }
300         }
301 
302         // Send back the flag...
303         return decryptResponse;
304     }
